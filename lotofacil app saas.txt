import streamlit as st
import pandas as pd
import numpy as np
import random
from sklearn.ensemble import RandomForestClassifier

st.title("🤖 Lotofácil AI SaaS")

file = st.file_uploader("Upload dos resultados (.csv)")

if file:
    dados = pd.read_csv(file)

    def transformar(dados):
        X, y = [], []
        for i in range(len(dados)-1):
            atual = dados.iloc[i]
            prox = dados.iloc[i+1]
            vetor = [1 if n in atual.values else 0 for n in range(1,26)]
            alvo = [1 if n in prox.values else 0 for n in range(1,26)]
            X.append(vetor)
            y.append(alvo)
        return np.array(X), np.array(y)

    X, y = transformar(dados)

    modelos = []
    for i in range(25):
        clf = RandomForestClassifier()
        clf.fit(X, y[:,i])
        modelos.append(clf)

    ultimo = X[-1].reshape(1,-1)

    probs = []
    for i, m in enumerate(modelos):
        p = m.predict_proba(ultimo)[0][1]
        probs.append((i+1, p))

    probs.sort(key=lambda x: x[1], reverse=True)

    st.subheader("📊 Ranking")
    st.write(probs)

    def gerar_jogo():
        base = [n for n,_ in probs[:20]]
        return sorted(random.sample(base, 15))

    st.subheader("🎯 Jogos")
    for _ in range(5):
        st.write(gerar_jogo())
